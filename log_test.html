create record

record := BasicLogRecord{size, nanos, index, flags, buf} assert.Equal(t, index, record.Index(), "index should be 0") assert.Equal(t, size, record.Size(), "size should be 64") assert.Equal(t, nanos, record.Time()) assert.Equal(t, flags, record.Flags()) assert.Equal(t, buf, record.Data()) } func TestBasicLogRecordFactory(t *testing.T) { var index uint64 buf := make([]byte, 64) var size uint32 = 64 nanos := time.Now().UnixNano() flags := DefaultRecordFlags

create record

create record

create record

test index

test size

s, err := xbinary.LittleEndian.Uint32(bin, 0) assert.Nil(t, err) assert.Equal(t, record.Size(), s, "size should be 64")

test time

test flags

f, err := xbinary.LittleEndian.Uint32(bin, 4) assert.Nil(t, err) assert.Equal(t, record.Flags(), f) } func TestBasicLogRecordUnmarshal(t *testing.T) { var index uint64 buf := make([]byte, 64) var size uint32 = 64 nanos := time.Now().UnixNano() flags := DefaultRecordFlags

test index

test flags
